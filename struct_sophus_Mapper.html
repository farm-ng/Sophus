template struct sophus::Mapper¶

Helper for mapping tangent vectors (scalars) over pointers to data.

#include <ceres_typetraits.hpp>

template <typename T, typename E = void>
struct Mapper {
    // typedefs

    using Scalar = T ;
    using Map = Scalar& ;
    using ConstMap = const Scalar& ;

    // methods

    static Map map(Scalar* ptr);
    static ConstMap map(const Scalar* ptr);
};
