Overview¶

Specialization of Eigen::Map for Sim2 const; derived from Sim2Base. More…

#include <sim2.hpp>

template <class ScalarT, int Options>
class Map<sophus::Sim2<ScalarT> const, Options>: public sophus::Sim2Base {
public:
    // typedefs

    using Base = sophus::Sim2Base<Map<sophus::Sim2<ScalarT> const, Options>> ;
    using Scalar = ScalarT ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar const* coeffs);

    // methods

    Map<sophus::RxSO2<Scalar> const, Options> const& rxso2() const;
    Map<sophus::Vector2<Scalar> const, Options> const& translation() const;

protected:
    // fields

    const Map<sophus::RxSO2<Scalar> const, Options> rxso2_;
    const Map<sophus::Vector2<Scalar> const, Options> translation_;
};

Detailed Documentation¶

Specialization of Eigen::Map for Sim2 const; derived from Sim2Base.

Allows us to wrap RxSO2 objects around POD array.
