Overview¶

Specialization of Eigen::Map for SE2; derived from SE2Base. More…

#include <se2.hpp>

template <class ScalarT, int Options>
class Map<sophus::SE2<ScalarT>, Options>: public sophus::SE2Base {
public:
    // typedefs

    using Base = sophus::SE2Base<Map<sophus::SE2<ScalarT>, Options>> ;
    using Scalar = ScalarT ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar* coeffs);

    // methods

    Map<sophus::SO2<Scalar>, Options>& so2();
    Map<sophus::SO2<Scalar>, Options> const& so2() const;
    Map<sophus::Vector2<Scalar>, Options>& translation();
    Map<sophus::Vector2<Scalar>, Options> const& translation() const;

protected:
    // fields

    Map<sophus::SO2<Scalar>, Options> so2_;
    Map<sophus::Vector2<Scalar>, Options> translation_;
};

Allows us to wrap SE2 objects around POD array.
