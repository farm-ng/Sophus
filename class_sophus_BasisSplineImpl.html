template class sophus::BasisSplineImpl¶

#include <spline.hpp>

template <class LieGroup_>
class BasisSplineImpl {
public:
    // typedefs

    using LieGroup = LieGroup_ ;
    using Scalar = typename LieGroup::Scalar ;
    using Transformation = typename LieGroup::Transformation ;
    using Tangent = typename LieGroup::Tangent ;

    // construction

    BasisSplineImpl(
        const std::vector<LieGroup>& parent_Ts_control_point,
        double delta_t
        );

    // methods

    LieGroup parent_T_spline(
        int i,
        double u
        ) const;

    Transformation Dt_parent_T_spline(
        int i,
        double u
        ) const;

    Transformation Dt2_parent_T_spline(
        int i,
        double u
        ) const;

    const std::vector<LieGroup>& parent_Ts_control_point() const;
    std::vector<LieGroup>& parent_Ts_control_point();
    int getNumSegments() const;
    double delta_t() const;
};
