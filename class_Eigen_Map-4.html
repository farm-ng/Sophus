Overview¶

Specialization of Eigen::Map for SE3; derived from SE3Base. More…

#include <se3.hpp>

template <class ScalarT, int Options>
class Map<sophus::SE3<ScalarT>, Options>: public sophus::SE3Base {
public:
    // typedefs

    using Base = sophus::SE3Base<Map<sophus::SE3<ScalarT>, Options>> ;
    using Scalar = ScalarT ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar* coeffs);

    // methods

    Map<sophus::SO3<Scalar>, Options>& so3();
    Map<sophus::SO3<Scalar>, Options> const& so3() const;
    Map<sophus::Vector3<Scalar, Options>>& translation();
    Map<sophus::Vector3<Scalar, Options>> const& translation() const;

protected:
    // fields

    Map<sophus::SO3<Scalar>, Options> so3_;
    Map<sophus::Vector3<Scalar>, Options> translation_;
};

Detailed Documentation¶

Specialization of Eigen::Map for SE3; derived from SE3Base.

Allows us to wrap SE3 objects around POD array.
