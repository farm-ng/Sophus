Overview¶

Specialization of Eigen::Map for Sim3; derived from Sim3Base. More…

#include <sim3.hpp>

template <class ScalarT, int Options>
class Map<sophus::Sim3<ScalarT>, Options>: public sophus::Sim3Base {
public:
    // typedefs

    using Base = sophus::Sim3Base<Map<sophus::Sim3<ScalarT>, Options>> ;
    using Scalar = ScalarT ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar* coeffs);

    // methods

    Map<sophus::RxSO3<Scalar>, Options>& rxso3();
    Map<sophus::RxSO3<Scalar>, Options> const& rxso3() const;
    Map<sophus::Vector3<Scalar>, Options>& translation();
    Map<sophus::Vector3<Scalar>, Options> const& translation() const;

protected:
    // fields

    Map<sophus::RxSO3<Scalar>, Options> rxso3_;
    Map<sophus::Vector3<Scalar>, Options> translation_;
};

Detailed Documentation¶

Specialization of Eigen::Map for Sim3; derived from Sim3Base.

Allows us to wrap Sim3 objects around POD array.
